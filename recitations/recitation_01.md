## 4. –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –ø–æ–∏—Å–∫–∞ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π –¥–Ω–µ–π —Ä–æ–∂–¥–µ–Ω–∏—è

### –û—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º (–∫–≤–∞–¥—Ä–∞—Ç–∏—á–Ω—ã–π O(n¬≤)):
```python
class StaticArray:
    def __init__(self, n):
        self.data = [None] * n
    def get_at(self, i):
        if not (0 <= i < len(self.data)): raise IndexError
        return self.data[i]
    def set_at(self, i, x):
        if not (0 <= i < len(self.data)): raise IndexError
        self.data[i] = x

def birthday_match(students):
    n = len(students)
    record = StaticArray(n)
    for k in range(n):
        name1, bday1 = students[k]
        for i in range(k):
            name2, bday2 = record.get_at(i)
            if bday1 == bday2:
                return (name1, name2)
        record.set_at(k, (name1, bday1))
    return None
```
### –û–ø—Ç–∏–º–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –≤–∞—Ä–∏–∞–Ω—Ç (–ª–∏–Ω–µ–π–Ω—ã–π O(n) —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º —Ö—ç—à-—Ç–∞–±–ª–∏—Ü—ã):

```python
def birthday_match_optimized(students):
    seen = {}
    for name, bday in students:
        if bday in seen:
            return (name, seen[bday])
        seen[bday] = name
    return None
```
